272:   def Feed.normalize_url(url)
273:     if url.kind_of?(URI)
274:       url = url.to_s
275:     end
276:     if url.blank?
277:       return nil
278:     end
279:     normalized_url = CGI.unescape(url.strip)
280: 
281:     # if a url begins with the '/' character, it only makes sense that they
282:     # meant to be using a file:// url.  Fix it for them.
283:     if normalized_url.length > 0 && normalized_url[0..0] == "/"
284:       normalized_url = "file://" + normalized_url
285:     end
286: 
287:     # if a url begins with a drive letter followed by a colon, we're looking at
288:     # a file:// url.  Fix it for them.
289:     if normalized_url.length > 0 &&
290:         normalized_url.scan(/^[a-zA-Z]:[\\\/]/).size > 0
291:       normalized_url = "file:///" + normalized_url
292:     end
293: 
294:     # if a url begins with javascript:, it's quite possibly an attempt at
295:     # doing something malicious.  Let's keep that from getting anywhere,
296:     # shall we?
297:     if (normalized_url.downcase =~ /javascript:/) != nil
298:       return "#"
299:     end
300: 
301:     # deal with all of the many ugly possibilities involved in the rss:
302:     # and feed: pseudo-protocols (incidentally, whose crazy idea was this
303:     # mess?)
304:     normalized_url.gsub!(/^http:\/*(feed:\/*)?/i, "http://")
305:     normalized_url.gsub!(/^http:\/*(rss:\/*)?/i, "http://")
306:     normalized_url.gsub!(/^feed:\/*(http:\/*)?/i, "http://")
307:     normalized_url.gsub!(/^rss:\/*(http:\/*)?/i, "http://")
308:     normalized_url.gsub!(/^file:\/*/i, "file:///")
309:     normalized_url.gsub!(/^https:\/*/i, "https://")
310:     # fix (very) bad urls (usually of the user-entered sort)
311:     normalized_url.gsub!(/^http:\/*(http:\/*)*/i, "http://")
312: 
313:     if (normalized_url =~ /^file:/i) == 0
314:       # Adjust windows-style urls
315:       normalized_url.gsub!(/^file:\/\/\/([a-zA-Z])\|/i, 'file:///\1:')
316:       normalized_url.gsub!(/\\/, '/')
317:     else
318:       if (normalized_url =~ /^https?:\/\//i) == nil
319:         normalized_url = "http://" + normalized_url
320:       end
321:       if normalized_url == "http://"
322:         return nil
323:       end
324:       begin
325:         scheme, host_part, path =
326:           normalized_url.scan(/^(https?):\/\/([^\/]+)\/(.*)/i).flatten
327:         if scheme != nil && host_part != nil && path != nil
328:           scheme = scheme.downcase
329:           if FeedTools::UriHelper.idn_enabled?
330:             host_part =
331:               IDN::Idna.toASCII(host_part)
332:           end
333:           new_path = ""
334:           for index in 0...path.size
335:             if path[index] <= 32 || path[index] >= 126
336:               new_path << ("%" + path[index].to_s(16).upcase)
337:             else
338:               new_path << path[index..index]
339:             end
340:           end
341:           path = new_path
342:           normalized_url = scheme + "://" + host_part + "/" + path
343:         end
344:       rescue Object
345:       end
346:       begin
347:         feed_uri = URI.parse(normalized_url)
348:         if feed_uri.scheme == nil
349:           feed_uri.scheme = "http"
350:         end
351:         if feed_uri.path.blank?
352:           feed_uri.path = "/"
353:         end
354:         if (feed_uri.path =~ /^[\/]+/) == 0
355:           feed_uri.path.gsub!(/^[\/]+/, "/")
356:         end
357:         while (feed_uri.path =~ /^\/\.\./)
358:           feed_uri.path.gsub!(/^\/\.\./, "")
359:         end
360:         if feed_uri.path.blank?
361:           feed_uri.path = "/"
362:         end
363:         feed_uri.host.downcase!
364:         normalized_url = feed_uri.to_s
365:       rescue URI::InvalidURIError
366:       end
367:     end
368: 
369:     # We can't do a proper set of escaping, so this will
370:     # have to do.
371:     normalized_url.gsub!(/%20/, " ")
372:     normalized_url.gsub!(/ /, "%20")
373: 
374:     return normalized_url
375:   end
